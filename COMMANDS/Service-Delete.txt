# You can use WMI or other tools for this since there is no Remove-Service cmdlet until Powershell 6.0 

# PowerShell 
Remove-Service -Name ServiceName 

 
# WMI (PowerShell) 
$service = Get-WmiObject -Class Win32_Service -Filter "Name='servicename'" 
$service.delete() 

 
# Command-Linem (CMD)
sc.exe delete ServiceName 
sc.exe \\server delete "MyService"
